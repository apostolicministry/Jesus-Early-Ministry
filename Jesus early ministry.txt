<script>
document.querySelectorAll('details.verse').forEach(box=>{
  const out=box.querySelector('.scripture');
  const ref=box.dataset.ref;
  box.addEventListener('toggle',async()=>{
    if(box.open && !out.textContent.trim()){
      out.innerHTML='<span class="loading">Loading…</span>';
      try{
        const r=await fetch('https://bible-api.com/'+encodeURIComponent(ref)+'?translation=web');
        const d=await r.json();
        out.innerHTML='<strong>'+d.reference+'</strong><br>'+d.text.replace(/\n/g,'<br>');
      }catch(e){out.textContent='Sorry – couldn’t load.';}
    }
  });
});
</script>
